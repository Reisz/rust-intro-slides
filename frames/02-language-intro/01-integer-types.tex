\begin{luacode*}
	function sizeList()
		local list = {8, 16, 32, 64, 128}

		for i, v in ipairs(list) do
			list[i] = ([[\texttt{%d}]]):format(v)
		end

		tex.sprint(table.concat(list, ", "))
	end
\end{luacode*}
